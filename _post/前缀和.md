#include<iostream>
using namespace std;
const int N = 1e5 + 10;

int n, m, q[N];

int main()
{
    scanf("%d%d", &n, &m);
    scanf("%d", &q[1]);
    for (int i = 2; i <= n; i ++)
    {
        int x;
        scanf("%d", &x);
        q[i] = q[i - 1] + x;
    }

    for (int i = 1; i <= m; i ++)
    {
        int l, r;
        scanf("%d%d",&l, &r);
        printf ("%d\n", q[r] - q[l - 1]);
    }
    return 0;
}
